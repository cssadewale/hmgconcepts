📋 Supabase Setup Instructions
Run these SQL commands in your Supabase SQL Editor to set up the database tables: 
-- ENQUIRIES TABLE
CREATE TABLE enquiries (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text NOT NULL,
  email text,
  phone text,
  role text,
  message text,
  read boolean DEFAULT false,
  created_at timestamptz DEFAULT now()
);

-- ANALYTICS TABLE
CREATE TABLE analytics (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  page text,
  device text,
  country text,
  referrer text,
  session_id text,
  created_at timestamptz DEFAULT now()
);

-- Enable Row Level Security
ALTER TABLE enquiries ENABLE ROW LEVEL SECURITY;
ALTER TABLE analytics ENABLE ROW LEVEL SECURITY;

-- Allow public insert for enquiries & analytics
CREATE POLICY "allow_insert" ON enquiries FOR INSERT WITH CHECK (true);
CREATE POLICY "allow_insert" ON analytics FOR INSERT WITH CHECK (true);

-- Allow authenticated read
CREATE POLICY "allow_read" ON enquiries FOR SELECT USING (auth.role() = 'authenticated');
CREATE POLICY "allow_read" ON analytics FOR SELECT USING (auth.role() = 'authenticated');
CREATE POLICY "allow_update" ON enquiries FOR UPDATE USING (auth.role() = 'authenticated');
CREATE POLICY "allow_delete" ON enquiries FOR DELETE USING (auth.role() = 'authenticated');
